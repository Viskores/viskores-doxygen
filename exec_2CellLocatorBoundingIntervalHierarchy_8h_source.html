126  VISKORES_EXEC viskores::ErrorCode FindCell(const viskores::Vec3f& point,
127  viskores::Id& cellId,
128  viskores::Vec3f& pCoords,
129  LastCell& lastCell) const
130  {
131  cellId = -1;
132 
133  //Check the last cell.
134  if ((lastCell.CellId >= 0) && (lastCell.CellId < this->CellSet.GetNumberOfElements()))
135  {
136  if (this->PointInCell(point, lastCell.CellId, pCoords))
137  {
138  cellId = lastCell.CellId;
139  return viskores::ErrorCode::Success;
140  }
141  }
142 
143  //Check the last leaf node.
144  if ((lastCell.NodeIdx >= 0) && (lastCell.NodeIdx < this->Nodes.GetNumberOfValues()))
145  {
146  const auto& node = this->Nodes.Get(lastCell.NodeIdx);
147 
148  if (node.ChildIndex < 0)
149  {
150  viskores::IdComponent count = 0;
151  viskores::Vec<viskores::Id, 1> cellIdsVec = { -1 };
152  viskores::Vec<viskores::Vec3f, 1> pCoordsVec = { viskores::Vec3f() };
153  if (this->FindInLeaf(IterateMode::FindOne, point, pCoordsVec, node, cellIdsVec, count))
154  {
155  lastCell.CellId = cellIdsVec[0];
156  pCoords = pCoordsVec[0];
157  return viskores::ErrorCode::Success;
158  }
159  }
160  }
161 
162  //No fastpath. Do a full search.
163  viskores::Vec<viskores::Id, 1> cellIdsVec = { -1 };
164  viskores::Vec<viskores::Vec3f, 1> pCoordsVec = { viskores::Vec3f() };
165 
166  viskores::IdComponent count = 0;
167  auto status =
168  this->FindCellImpl(IterateMode::FindOne, point, cellIdsVec, pCoordsVec, lastCell, count);
169 
170  cellId = cellIdsVec[0];
171  pCoords = pCoordsVec[0];
172  return status;
173  }

244  template <typename CellIdsType, typename ParametricCoordsVecType>
245  VISKORES_EXEC viskores::ErrorCode FindCellImpl(const IterateMode& mode,
246  const viskores::Vec3f& point,
247  CellIdsType& cellIdsVec,
248  ParametricCoordsVecType& pCoordsVec,
249  LastCell& lastCell,
250  viskores::IdComponent& count) const
251  {
252  viskores::Id nodeIndex = 0;
253  FindCellState state = FindCellState::EnterNode;
254  VISKORES_ASSERT(cellIdsVec.GetNumberOfComponents() > 0);
255 
256  cellIdsVec[0] = -1;
257 
258  VISKORES_ASSERT(this->Nodes.GetNumberOfValues() > 0);
259  while (true)
260  {
261  // 1) If we’ve found a cell (and only looking for one), stop immediately
262  if (mode == IterateMode::FindOne && cellIdsVec[0] > viskores::Id(0))
263  break;
264 
265  // 2) If we’ve returned all the way to the root and just ascended, stop
266  if (nodeIndex == 0 && state == FindCellState::AscendFromNode)
267  break;
268 
269  // 3) Otherwise, do exactly one step of the state machine
270  switch (state)
271  {
272  case FindCellState::EnterNode:
273  state = this->EnterNode(mode, point, cellIdsVec, nodeIndex, pCoordsVec, lastCell, count);
274  break;
275 
276  case FindCellState::AscendFromNode:
277  state = this->AscendFromNode(nodeIndex);
278  break;
279 
280  case FindCellState::DescendLeftChild:
281  state = this->DescendLeftChild(point, nodeIndex);
282  break;
283 
284  case FindCellState::DescendRightChild:
285  state = this->DescendRightChild(point, nodeIndex);
286  break;
287  }
288  }
289 
290  if (count == 0)
291  return viskores::ErrorCode::CellNotFound;
292 
293  return viskores::ErrorCode::Success;
294  }

296  template <typename CellIdsType, typename ParametricCoordsVecType>
297  VISKORES_EXEC FindCellState EnterNode(const IterateMode& mode,
298  const viskores::Vec3f& point,
299  CellIdsType& cellIdsVec,
300  viskores::Id nodeIndex,
301  ParametricCoordsVecType& pCoordsVec,
302  LastCell& lastCell,
303  viskores::IdComponent& count) const
304  {
305  viskores::IdComponent n = cellIdsVec.GetNumberOfComponents();
306  VISKORES_ASSERT(pCoordsVec.GetNumberOfComponents() == n);
307 
308  const viskores::exec::CellLocatorBoundingIntervalHierarchyNode& node =
309  this->Nodes.Get(nodeIndex);
310 
311  if (node.ChildIndex < 0)
312  {
313  // In a leaf node. Look for a containing cell.
314  if (this->FindInLeaf(mode, point, pCoordsVec, node, cellIdsVec, count))
315  {
316  lastCell.NodeIdx = nodeIndex;
317  lastCell.CellId = cellIdsVec[0];
318  }
319 
320  return FindCellState::AscendFromNode;
321  }
322  else
323  {
324  return FindCellState::DescendLeftChild;
325  }
326  }

328  VISKORES_EXEC
329  FindCellState AscendFromNode(viskores::Id& nodeIndex) const
330  {
331  viskores::Id childNodeIndex = nodeIndex;
332  const viskores::exec::CellLocatorBoundingIntervalHierarchyNode& childNode =
333  this->Nodes.Get(childNodeIndex);
334  nodeIndex = childNode.ParentIndex;
335  const viskores::exec::CellLocatorBoundingIntervalHierarchyNode& parentNode =
336  this->Nodes.Get(nodeIndex);
337 
338  if (parentNode.ChildIndex == childNodeIndex)
339  {
340  // Ascending from left child. Descend into the right child.
341  return FindCellState::DescendRightChild;
342  }
343  else
344  {
345  VISKORES_ASSERT(parentNode.ChildIndex + 1 == childNodeIndex);
346  // Ascending from right child. Ascend again. (Don't need to change state.)
347  return FindCellState::AscendFromNode;
348  }
349  }

351  VISKORES_EXEC
352  FindCellState DescendLeftChild(const viskores::Vec3f& point, viskores::Id& nodeIndex) const
353  {
354  const viskores::exec::CellLocatorBoundingIntervalHierarchyNode& node =
355  this->Nodes.Get(nodeIndex);
356  VISKORES_ASSERT(node.Dimension >= 0 && node.Dimension < 3);
357 
358  const viskores::FloatDefault& coordinate = point[node.Dimension];
359  if (coordinate <= node.Node.LMax)
360  {
361  // Left child does contain the point. Do the actual descent.
362  nodeIndex = node.ChildIndex;
363  return FindCellState::EnterNode;
364  }
365  else
366  {
367  // Left child does not contain the point. Skip to the right child.
368  return FindCellState::DescendRightChild;
369  }
370  }

372  VISKORES_EXEC
373  FindCellState DescendRightChild(const viskores::Vec3f& point, viskores::Id& nodeIndex) const
374  {
375  const viskores::exec::CellLocatorBoundingIntervalHierarchyNode& node =
376  this->Nodes.Get(nodeIndex);
377  VISKORES_ASSERT(node.Dimension >= 0 && node.Dimension < 3);
378 
379  const viskores::FloatDefault& coordinate = point[node.Dimension];
380  if (coordinate >= node.Node.RMin)
381  {
382  // Right child does contain the point. Do the actual descent.
383  nodeIndex = node.ChildIndex + 1;
384  return FindCellState::EnterNode;

388  // Right child does not contain the point. Skip to ascent
389  return FindCellState::AscendFromNode;

393  template <typename CellIdsType, typename ParametricCoordsVecType>
394  VISKORES_EXEC bool FindInLeaf(
395  const IterateMode& mode,
396  const viskores::Vec3f& point,
397  ParametricCoordsVecType& pCoordsVec,
398  const viskores::exec::CellLocatorBoundingIntervalHierarchyNode& node,
399  CellIdsType& cellIdsVec,
400  viskores::IdComponent& count) const
401  {
402  auto n = cellIdsVec.GetNumberOfComponents();
403 
404  bool found = false;
405 
406  for (viskores::Id i = node.Leaf.Start; i < node.Leaf.Start + node.Leaf.Size; ++i)
407  {
408  viskores::Id cid = this->CellIds.Get(i);
409  viskores::Vec3f pCoords;
410  if (this->PointInCell(point, cid, pCoords))
411  {
412  found = true;
413  if (mode == IterateMode::FindOne || mode == IterateMode::FindAll)
414  {
415  VISKORES_ASSERT(count < n);
416  //Update vecs if there is room. If the vecs are too small, it will get reported as an error in FindAllCells()
417  if (count < n)
418  {
419  cellIdsVec[count] = cid;
420  pCoordsVec[count] = pCoords;
421  }
422  }
423  count++;
424 
425  if (mode == IterateMode::FindOne)
426  break;
427  }
428  }
429  return found;
430  }

432  VISKORES_EXEC bool PointInCell(const viskores::Vec3f& point,
433  viskores::Id& cellId,
434  viskores::Vec3f& pCoords) const
435  {
436  using IndicesType = typename CellSetPortal::IndicesType;
437  IndicesType cellPointIndices = this->CellSet.GetIndices(cellId);
438  viskores::VecFromPortalPermute<IndicesType, CoordsPortal> cellPoints(&cellPointIndices,
439  this->Coords);
440  auto cellShape = this->CellSet.GetCellShape(cellId);
441  auto status = viskores::exec::WorldCoordinatesToParametricCoordinates(
442  cellPoints, point, cellShape, pCoords);
443 
444  if (status != viskores::ErrorCode::Success)
445  return false;
446 
447  if (!viskores::exec::CellInside(pCoords, cellShape))
448  return false;
449 
450  return true;
451  }
