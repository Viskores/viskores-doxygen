76 template <typename PointsVecType>
77 VISKORES_EXEC inline Bounds ComputeCellBounds(const PointsVecType& points)
78 {
79  using CoordsType = typename viskores::VecTraits<PointsVecType>::ComponentType;
80  auto numPoints = viskores::VecTraits<PointsVecType>::GetNumberOfComponents(points);
81 
82  CoordsType minp = points[0], maxp = points[0];
83  for (viskores::IdComponent i = 1; i < numPoints; ++i)
84  {
85  minp = viskores::Min(minp, points[i]);
86  maxp = viskores::Max(maxp, points[i]);
87  }
88 
89  return { FloatVec3(minp), FloatVec3(maxp) };
90 }
91 }

123  // TODO: This function may return false positives for non 3D cells as the
124  // tests are done on the projection of the point on the cell. Extra checks
125  // should be added to test if the point actually falls on the cell.
126  template <typename CellShapeTag, typename CoordsType>
127  VISKORES_EXEC static viskores::ErrorCode PointInsideCell(FloatVec3 point,
128  CellShapeTag cellShape,
129  CoordsType cellPoints,
130  FloatVec3& pCoords,
131  bool& inside)
132  {
133  auto bounds = viskores::internal::cl_uniform_bins::ComputeCellBounds(cellPoints);
134  if (point[0] >= bounds.Min[0] && point[0] <= bounds.Max[0] && point[1] >= bounds.Min[1] &&
135  point[1] <= bounds.Max[1] && point[2] >= bounds.Min[2] && point[2] <= bounds.Max[2])
136  {
137  VISKORES_RETURN_ON_ERROR(viskores::exec::WorldCoordinatesToParametricCoordinates(
138  cellPoints, point, cellShape, pCoords));
139  inside = viskores::exec::CellInside(pCoords, cellShape);
140  }
141  else
142  {
143  inside = false;
144  }
145  // Return success error code even point is not inside this cell
146  return viskores::ErrorCode::Success;
147  }

193  VISKORES_EXEC
194  viskores::ErrorCode FindCell(const FloatVec3& point,
195  viskores::Id& cellId,
196  FloatVec3& pCoords,
197  LastCell& lastCell) const
198  {
199  viskores::Vec3f pc;
200  //See if point is inside the last cell.
201  if ((lastCell.CellId >= 0) && (lastCell.CellId < this->CellSet.GetNumberOfElements()) &&
202  this->PointInCell(point, lastCell.CellId, pc) == viskores::ErrorCode::Success)
203  {
204  pCoords = pc;
205  cellId = lastCell.CellId;
206  return viskores::ErrorCode::Success;
207  }
208 
209  //See if it's in the last leaf.
210  if ((lastCell.LeafIdx >= 0) && (lastCell.LeafIdx < this->CellCount.GetNumberOfValues()) &&
211  this->PointInLeaf(point, lastCell.LeafIdx, cellId, pc) == viskores::ErrorCode::Success)
212  {
213  pCoords = pc;
214  lastCell.CellId = cellId;
215  return viskores::ErrorCode::Success;
216  }
217 
218  //Call the full point search.
219  viskores::Vec<viskores::Id, 1> cellIdVec = { -1 };
220  viskores::Vec<viskores::Vec3f, 1> pCoordsVec;
221  auto nCells = this->IterateLeaves(point, IterateMode::FindOne, cellIdVec, pCoordsVec, lastCell);
222  if (nCells < 0)
223  {
224  cellId = -1;
225  return viskores::ErrorCode::InvalidNumberOfIndices;
226  }
227  else if (nCells == 0)
228  {
229  cellId = -1;
230  return viskores::ErrorCode::CellNotFound;
231  }
232  else
233  {
234  cellId = cellIdVec[0];
235  pCoords = pCoordsVec[0];
236  return viskores::ErrorCode::Success;
237  }
238  }

251  template <typename CellIdsType, typename ParametricCoordsVecType>
252  VISKORES_EXEC viskores::ErrorCode FindAllCells(const viskores::Vec3f& point,
253  CellIdsType& cellIdVec,
254  ParametricCoordsVecType& pCoordsVec) const
255  {
256  viskores::IdComponent n = cellIdVec.GetNumberOfComponents();
257  VISKORES_ASSERT(pCoordsVec.GetNumberOfComponents() == n);
258 
259  if (n == 0)
260  return viskores::ErrorCode::Success;
261 
262  for (viskores::IdComponent i = 0; i < n; i++)
263  cellIdVec[i] = -1;
264 
265  LastCell lastCell;
266  viskores::IdComponent nCells =
267  this->IterateLeaves(point, IterateMode::FindAll, cellIdVec, pCoordsVec, lastCell);
268  VISKORES_ASSERT(n == nCells);
269  if (nCells == 0)
270  return viskores::ErrorCode::CellNotFound;
271 
272  return viskores::ErrorCode::Success;
273  }

275 private:
276  // Shared leaf traversal: modes are FindOne, CountAll, FindAll
277  enum class IterateMode
278  {
279  FindOne,
280  CountAll,
281  FindAll
282  };
283 
284  template <typename CellIdVecType, typename ParametricCoordsVecType>
285  VISKORES_EXEC viskores::IdComponent IterateLeaves(const FloatVec3& point,
286  const IterateMode& mode,
287  CellIdVecType& cellIdVec,
288  ParametricCoordsVecType& pCoordsVec,
289  LastCell& lastCell) const
290  {
291  using namespace viskores::internal::cl_uniform_bins;
292 
293  viskores::IdComponent n = cellIdVec.GetNumberOfComponents();
294  VISKORES_ASSERT(pCoordsVec.GetNumberOfComponents() == n);
295 
296  lastCell.CellId = -1;
297  lastCell.LeafIdx = -1;
298  viskores::IdComponent cellCnt = 0;
299 
300  DimVec3 binId3 = static_cast<DimVec3>((point - this->TopLevel.Origin) / this->TopLevel.BinSize);
301  if (binId3[0] >= 0 && binId3[0] < this->TopLevel.Dimensions[0] && binId3[1] >= 0 &&
302  binId3[1] < this->TopLevel.Dimensions[1] && binId3[2] >= 0 &&
303  binId3[2] < this->TopLevel.Dimensions[2])
304  {
305  viskores::Id binId = ComputeFlatIndex(binId3, this->TopLevel.Dimensions);
306 
307  auto ldim = this->LeafDimensions.Get(binId);
308  if (!ldim[0] || !ldim[1] || !ldim[2])
309  {
310  return 0;
311  }
312 
313  auto leafGrid = ComputeLeafGrid(binId3, ldim, this->TopLevel);
314 
315  DimVec3 leafId3 = static_cast<DimVec3>((point - leafGrid.Origin) / leafGrid.BinSize);
316  // precision issues may cause leafId3 to be out of range so clamp it
317  leafId3 = viskores::Max(DimVec3(0), viskores::Min(ldim - DimVec3(1), leafId3));
318 
319  viskores::Id leafStart = this->LeafStartIndex.Get(binId);
320  viskores::Id leafIdx = leafStart + ComputeFlatIndex(leafId3, leafGrid.Dimensions);
321 
322  viskores::Id start = this->CellStartIndex.Get(leafIdx);
323  viskores::Id end = start + this->CellCount.Get(leafIdx);
324  for (viskores::Id i = start; i < end; ++i)
325  {
326  viskores::Id cellId = this->CellIds.Get(i);
327  viskores::Vec3f pc;
328  if (this->PointInCell(point, cellId, pc) == viskores::ErrorCode::Success)
329  {
330  lastCell.CellId = cellId;
331  lastCell.LeafIdx = leafIdx;
332  if (mode != IterateMode::CountAll)
333  {
334  cellIdVec[cellCnt] = cellId;
335  pCoordsVec[cellCnt] = pc;
336  }
337  cellCnt++;
338  if (mode == IterateMode::FindOne)
339  break;
340  }
341  }
342  }
343 
344  return static_cast<viskores::Id>(cellCnt);
345  }

347  VISKORES_EXEC viskores::ErrorCode PointInCell(const viskores::Vec3f& point,
348  const viskores::Id& cid,
349  viskores::Vec3f& pCoords) const
350  {
351  auto indices = this->CellSet.GetIndices(cid);
352  auto pts = viskores::make_VecFromPortalPermute(&indices, this->Coords);
353  viskores::Vec3f pc;
354  bool inside;
355  auto status = PointInsideCell(point, this->CellSet.GetCellShape(cid), pts, pc, inside);
356  if (status == viskores::ErrorCode::Success && inside)
357  {
358  pCoords = pc;
359  return viskores::ErrorCode::Success;
360  }
361 
362  return viskores::ErrorCode::CellNotFound;
363  }
