76 template <typename PointsVecType>
77 VISKORES_EXEC inline Bounds ComputeCellBounds(const PointsVecType& points)
78 {
79  using CoordsType = typename viskores::VecTraits<PointsVecType>::ComponentType;
80  auto numPoints = viskores::VecTraits<PointsVecType>::GetNumberOfComponents(points);
81 
82  CoordsType minp = points[0], maxp = points[0];
83  for (viskores::IdComponent i = 1; i < numPoints; ++i)
84  {
85  minp = viskores::Min(minp, points[i]);
86  maxp = viskores::Max(maxp, points[i]);
87  }
88 
89  return { FloatVec3(minp), FloatVec3(maxp) };
90 }

123  // TODO: This function may return false positives for non 3D cells as the
124  // tests are done on the projection of the point on the cell. Extra checks
125  // should be added to test if the point actually falls on the cell.
126  template <typename CellShapeTag, typename CoordsType>
127  VISKORES_EXEC static viskores::ErrorCode PointInsideCell(FloatVec3 point,
128  CellShapeTag cellShape,
129  CoordsType cellPoints,
130  FloatVec3& pCoords,
131  bool& inside)
132  {
133  auto bounds = viskores::internal::cl_uniform_bins::ComputeCellBounds(cellPoints);
134  if (point[0] >= bounds.Min[0] && point[0] <= bounds.Max[0] && point[1] >= bounds.Min[1] &&
135  point[1] <= bounds.Max[1] && point[2] >= bounds.Min[2] && point[2] <= bounds.Max[2])
136  {
137  VISKORES_RETURN_ON_ERROR(viskores::exec::WorldCoordinatesToParametricCoordinates(
138  cellPoints, point, cellShape, pCoords));
139  inside = viskores::exec::CellInside(pCoords, cellShape);
140  }
141  else
142  {
143  inside = false;
144  }
145  // Return success error code even point is not inside this cell
146  return viskores::ErrorCode::Success;
147  }

252  VISKORES_EXEC viskores::ErrorCode FindCellImpl(const FloatVec3& point,
253  viskores::Id& cellId,
254  LastCell& lastCell,
255  FloatVec3* pCoords) const
256  {
257  viskores::Vec3f pc;
258  if ((lastCell.CellId >= 0) && (lastCell.CellId < this->CellSet.GetNumberOfElements()) &&
259  this->PointInCell(point, lastCell.CellId, pc) == viskores::ErrorCode::Success)
260  {
261  cellId = lastCell.CellId;
262  if (pCoords != nullptr)
263  *pCoords = pc;
264  return viskores::ErrorCode::Success;
265  }
266 
267  if ((lastCell.LeafIdx >= 0) && (lastCell.LeafIdx < this->CellCount.GetNumberOfValues()) &&
268  this->PointInLeaf(point, lastCell.LeafIdx, cellId, pc) == viskores::ErrorCode::Success)
269  {
270  lastCell.CellId = cellId;
271  if (pCoords != nullptr)
272  *pCoords = pc;
273  return viskores::ErrorCode::Success;
274  }
275 
276  viskores::Vec<viskores::Id, 1> cellIdVec = { -1 };
277  viskores::IdComponent nCells = 0;
278  if (pCoords != nullptr)
279  {
280  viskores::Vec<viskores::Vec3f, 1> pCoordsVec;
281  nCells = this->IterateLeaves(point, IterateMode::FindOne, cellIdVec, pCoordsVec, lastCell);
282  if (nCells == 1)
283  *pCoords = pCoordsVec[0];
284  }
285  else
286  {
287  nCells = this->IterateLeavesCellIds(point, IterateMode::FindOne, cellIdVec, lastCell);
288  }
289 
290  if (nCells == 0)
291  {
292  cellId = -1;
293  return viskores::ErrorCode::CellNotFound;
294  }
295 
296  cellId = cellIdVec[0];
297  return viskores::ErrorCode::Success;
298  }

382  template <typename CellIdVecType, typename WriteHitType>
383  VISKORES_EXEC viskores::IdComponent IterateLeavesImpl(const FloatVec3& point,
384  const IterateMode& mode,
385  CellIdVecType& cellIdVec,
386  LastCell& lastCell,
387  WriteHitType& writeHit) const
388  {
389  using namespace viskores::internal::cl_uniform_bins;
390 
391  viskores::IdComponent n = cellIdVec.GetNumberOfComponents();
392  lastCell.CellId = -1;
393  lastCell.LeafIdx = -1;
394  viskores::IdComponent cellCnt = 0;
395 
396  DimVec3 binId3 = static_cast<DimVec3>((point - this->TopLevel.Origin) / this->TopLevel.BinSize);
397  if (binId3[0] >= 0 && binId3[0] < this->TopLevel.Dimensions[0] && binId3[1] >= 0 &&
398  binId3[1] < this->TopLevel.Dimensions[1] && binId3[2] >= 0 &&
399  binId3[2] < this->TopLevel.Dimensions[2])
400  {
401  viskores::Id binId = ComputeFlatIndex(binId3, this->TopLevel.Dimensions);
402 
403  auto ldim = this->LeafDimensions.Get(binId);
404  if (!ldim[0] || !ldim[1] || !ldim[2])
405  {
406  return 0;
407  }
408 
409  auto leafGrid = ComputeLeafGrid(binId3, ldim, this->TopLevel);
410 
411  DimVec3 leafId3 = static_cast<DimVec3>((point - leafGrid.Origin) / leafGrid.BinSize);
412  // precision issues may cause leafId3 to be out of range so clamp it
413  leafId3 = viskores::Max(DimVec3(0), viskores::Min(ldim - DimVec3(1), leafId3));
414 
415  viskores::Id leafStart = this->LeafStartIndex.Get(binId);
416  viskores::Id leafIdx = leafStart + ComputeFlatIndex(leafId3, leafGrid.Dimensions);
417 
418  viskores::Id start = this->CellStartIndex.Get(leafIdx);
419  viskores::Id end = start + this->CellCount.Get(leafIdx);
420  for (viskores::Id i = start; i < end; ++i)
421  {
422  viskores::Id cellId = this->CellIds.Get(i);
423  viskores::Vec3f pc;
424  if (this->PointInCell(point, cellId, pc) == viskores::ErrorCode::Success)
425  {
426  lastCell.CellId = cellId;
427  lastCell.LeafIdx = leafIdx;
428  if (mode != IterateMode::CountAll)
429  writeHit(cellCnt, cellId, pc);
430 
431  cellCnt++;
432  if (mode == IterateMode::FindOne || (mode == IterateMode::FindAll && cellCnt == n))
433  break;
434  }
435  }
436  }
437 
438  return cellCnt;
439  }

465  VISKORES_EXEC viskores::ErrorCode PointInCell(const viskores::Vec3f& point,
466  const viskores::Id& cid,
467  viskores::Vec3f& pCoords) const
468  {
469  auto indices = this->CellSet.GetIndices(cid);
470  auto pts = viskores::make_VecFromPortalPermute(&indices, this->Coords);
471  viskores::Vec3f pc;
472  bool inside;
473  auto status = PointInsideCell(point, this->CellSet.GetCellShape(cid), pts, pc, inside);
474  if (status == viskores::ErrorCode::Success && inside)
475  {
476  pCoords = pc;
477  return viskores::ErrorCode::Success;
478  }
479 
480  return viskores::ErrorCode::CellNotFound;
481  }
