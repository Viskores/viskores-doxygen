43  // This struct helps manage a set of parameters providing arrays of data that
44  // will be attached to fields on a Viskores dataset created by this geometry.
45  // The fields are in the form `association.name` where `association` is the
46  // string used to describe what type of element each value is attached to
47  // (i.e., `vertex`, `primitive`) and `name` is how ANARI references the array
48  // (i.e., `color`, `attribute0`). The structure will register itself with the
49  // given `Geometry` object so that the geometry will be updated on data
50  // change. Use `commitParameters` to pull the arrays from the associated
51  // parameters (usually done in a `commitParameters` method of the geometry
52  // object). Use `setFields` to add these arrays as fields to a provided
53  // `viskores::cont::DataSet` (usually done in a `finalize` method of the
54  // geometry object).
55  class FieldArrayParameters
56  {
57  Geometry* m_geometry;
58  std::map<std::string, helium::ChangeObserverPtr<Array1D>> m_attributes;
59  std::string m_anariAssociation;
60  viskores::cont::Field::Association m_viskoresAssociation;
61 
62  public:
63  void setAttributes(Geometry* self, std::initializer_list<const char*>&& attributes);
64  void setAnariAssociation(const std::string& association);
65  void setViskoresAssociation(viskores::cont::Field::Association association);
66 
67  void commitParameters();

69 
70  // Gets a specific named parameter (perhaps because it is special). The
71  // string should be the same as that given in the list provided to
72  // `setAttributes`. It will not have the association part included in
73  // ANARI's name. This method only works after `commitParameters` is called.
74  helium::ChangeObserverPtr<Array1D>& getParam(const std::string& attribute);
75  const helium::ChangeObserverPtr<Array1D>& getParam(const std::string& attribute) const;
