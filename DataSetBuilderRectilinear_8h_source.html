284  template <typename T>
285  VISKORES_CONT static viskores::cont::DataSet BuildDataSet(const viskores::cont::ArrayHandle<T>& X,
286  const viskores::cont::ArrayHandle<T>& Y,
287  const viskores::cont::ArrayHandle<T>& Z,
288  const std::string& coordNm)
289  {
290  viskores::cont::DataSet dataSet;
291 
292  //Convert all coordinates to floatDefault.
293  viskores::cont::ArrayHandleCartesianProduct<viskores::cont::ArrayHandle<viskores::FloatDefault>,
294  viskores::cont::ArrayHandle<viskores::FloatDefault>,
295  viskores::cont::ArrayHandle<viskores::FloatDefault>>
296  coords;
297 
298  viskores::cont::ArrayHandle<viskores::FloatDefault> Xc, Yc, Zc;
299  DataSetBuilderRectilinear::CopyInto(X, Xc);
300  DataSetBuilderRectilinear::CopyInto(Y, Yc);
301  DataSetBuilderRectilinear::CopyInto(Z, Zc);
302 
303  coords = viskores::cont::make_ArrayHandleCartesianProduct(Xc, Yc, Zc);
304  viskores::cont::CoordinateSystem cs(coordNm, coords);
305  dataSet.AddCoordinateSystem(cs);
306 
307  // compute the dimensions of the cellset by counting the number of axes
308  // with >1 dimension
309  int ndims = 0;
310  viskores::Id dims[3];
311  if (Xc.GetNumberOfValues() > 1)
312  {
313  dims[ndims++] = Xc.GetNumberOfValues();
314  }
315  if (Yc.GetNumberOfValues() > 1)
316  {
317  dims[ndims++] = Yc.GetNumberOfValues();
318  }
319  if (Zc.GetNumberOfValues() > 1)
320  {
321  dims[ndims++] = Zc.GetNumberOfValues();
322  }
323 
324  if (ndims == 1)
325  {
326  viskores::cont::CellSetStructured<1> cellSet;
327  cellSet.SetPointDimensions(dims[0]);
328  dataSet.SetCellSet(cellSet);
329  }
330  else if (ndims == 2)
331  {
332  viskores::cont::CellSetStructured<2> cellSet;
333  cellSet.SetPointDimensions(viskores::make_Vec(dims[0], dims[1]));
334  dataSet.SetCellSet(cellSet);
335  }
336  else if (ndims == 3)
337  {
338  viskores::cont::CellSetStructured<3> cellSet;
339  cellSet.SetPointDimensions(viskores::make_Vec(dims[0], dims[1], dims[2]));
340  dataSet.SetCellSet(cellSet);
341  }
342  else
343  {
344  throw viskores::cont::ErrorBadValue("Invalid cell set dimension");
345  }
346 
347  return dataSet;
348  }
