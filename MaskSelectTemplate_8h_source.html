72 template <typename ArrayHandleType>
73 viskores::worklet::MaskSelect::ThreadToOutputMapType MaskSelectBuild(
74  const ArrayHandleType& maskArray,
75  viskores::cont::DeviceAdapterId device)
76 {
77  viskores::cont::ArrayHandle<viskores::Id> outputToThreadMap;
78  viskores::Id numThreads = viskores::cont::Algorithm::ScanExclusive(
79  device, viskores::cont::make_ArrayHandleCast<viskores::Id>(maskArray), outputToThreadMap);
80  VISKORES_ASSERT(numThreads <= maskArray.GetNumberOfValues());
81 
82  // We have implemented two different ways to compute the thread to output map. The first way is
83  // to use a binary search on each thread index into the output map. The second way is to
84  // schedule on each output and copy the the index to the thread map. The first way is faster
85  // for output sizes that are small relative to the input and also tends to be well load
86  // balanced. The second way is faster for larger outputs.
87  //
88  // The former is obviously faster for one thread and the latter is obviously faster when all
89  // outputs have a thread. We have to guess for values in the middle. I'm using if the square of
90  // the number of threads is less than the number of outputs because it is easy to compute.
91  if (numThreads == maskArray.GetNumberOfValues())
92  { //fast path when everything is on
93  return BuildThreadToOutputMapAllOn(numThreads, device);
94  }
95  else if ((numThreads * numThreads) < maskArray.GetNumberOfValues())
96  {
97  return BuildThreadToOutputMapWithFind(numThreads, outputToThreadMap, device);
98  }
99  else
100  {
101  return BuildThreadToOutputMapWithCopy(numThreads, outputToThreadMap, maskArray, device);
102  }
103 }
