260  const std::vector<viskores::cont::internal::Buffer>& buffers,
261  viskores::cont::DeviceAdapterId device,
262  viskores::cont::Token& token)
263  {
264  viskores::Id numValues = GetNumberOfValues(buffers);
265  ReadPortalType portal;
266  for (viskores::IdComponent componentIndex = 0; componentIndex < NUM_COMPONENTS;
267  ++componentIndex)

272  if ((componentArray.GetModulo() > 0) && (componentArray.GetModulo() < numValues))

588 template <typename ValueType>
589 struct Serialization<viskores::cont::ArrayHandleSOAStride<ValueType>>
590 {
591  using BaseType = viskores::cont::ArrayHandle<ValueType, viskores::cont::StorageTagSOAStride>;
592  static constexpr viskores::IdComponent NUM_COMPONENTS =
593  viskores::VecTraits<ValueType>::NUM_COMPONENTS;
594  using ComponentType = typename viskores::VecTraits<ValueType>::ComponentType;
595 
596  static VISKORES_CONT void save(BinaryBuffer& bb, const BaseType& obj_)
597  {
598  viskores::cont::ArrayHandleSOAStride<ValueType> obj = obj_;
599  for (viskores::IdComponent componentIndex = 0; componentIndex < NUM_COMPONENTS;
600  ++componentIndex)
601  {
602  viskores::cont::ArrayHandleStride<ComponentType> componentArray =
603  obj.GetArray(componentIndex);
604  viskoresdiy::save(bb, componentArray.GetNumberOfValues());
605  viskoresdiy::save(bb, componentArray.GetStride());
606  viskoresdiy::save(bb, componentArray.GetOffset());
607  viskoresdiy::save(bb, componentArray.GetModulo());
608  viskoresdiy::save(bb, componentArray.GetDivisor());
609 
610  viskores::cont::internal::Buffer componentBuffer = componentArray.GetBuffers()[1];
611  bool alreadyWritten = false;
612  for (viskores::IdComponent checkIndex = 0; checkIndex < componentIndex; ++checkIndex)
613  {
614  viskores::cont::internal::Buffer checkBuffer = obj.GetArray(checkIndex).GetBuffers()[1];
615  if (checkBuffer == componentBuffer)
616  {
617  viskoresdiy::save(bb, checkIndex);
618  alreadyWritten = true;
619  break;
620  }
621  }
622  if (!alreadyWritten)
623  {
624  viskoresdiy::save(bb, viskores::IdComponent{ -1 });
625  viskoresdiy::save(bb, componentBuffer);
626  }
627  }
628  }

630  static VISKORES_CONT void load(BinaryBuffer& bb, BaseType& obj_)
631  {
632  viskores::cont::ArrayHandleSOAStride<ValueType> obj;
633  for (std::size_t componentIndex = 0; componentIndex < NUM_COMPONENTS; ++componentIndex)
634  {
635  viskores::Id numValues;
636  viskores::Id stride;
637  viskores::Id offset;
638  viskores::Id modulo;
639  viskores::Id divisor;
640  viskores::cont::internal::Buffer componentBuffer;
641 
642  viskoresdiy::load(bb, numValues);
643  viskoresdiy::load(bb, stride);
644  viskoresdiy::load(bb, offset);
645  viskoresdiy::load(bb, modulo);
646  viskoresdiy::load(bb, divisor);
647 
648  viskores::IdComponent bufferIndex;
649  viskoresdiy::load(bb, bufferIndex);
650  if (bufferIndex < 0)
651  {
652  viskoresdiy::load(bb, componentBuffer);
653  }
654  else
655  {
656  componentBuffer = obj.GetArray(bufferIndex).GetBuffers()[1];
657  }
658 
659  viskores::cont::ArrayHandleStride<ComponentType> componentArray(
660  componentBuffer, numValues, stride, offset, modulo, divisor);
661  obj.SetArray(componentIndex, componentArray);
662  }
663  obj_ = obj;
664  }
