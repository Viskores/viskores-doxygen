67  VISKORES_CONT static std::vector<viskores::cont::internal::Buffer> CreateBuffers()
68  {
69  std::vector<viskores::cont::internal::Buffer> buffers;
70  buffers.reserve(static_cast<std::size_t>(NUM_COMPONENTS * NUM_BUFFERS_PER_COMPONENT));
71  for (viskores::IdComponent i = 0; i < NUM_COMPONENTS; ++i)
72  {
73  ComponentArrayType newArray;
74  std::vector<viskores::cont::internal::Buffer> newBuffers = newArray.GetBuffers();
75  VISKORES_ASSERT(newBuffers.size() == static_cast<std::size_t>(NUM_BUFFERS_PER_COMPONENT));
76  buffers.insert(buffers.end(), newBuffers.begin(), newBuffers.end());
77  }
78  return buffers;
79  }

466 template <typename ValueType>
467 struct Serialization<viskores::cont::ArrayHandleSOAStride<ValueType>>
468 {
469  using BaseType = viskores::cont::ArrayHandle<ValueType, viskores::cont::StorageTagSOAStride>;
470  static constexpr viskores::IdComponent NUM_COMPONENTS =
471  viskores::VecTraits<ValueType>::NUM_COMPONENTS;
472  using ComponentType = typename viskores::VecTraits<ValueType>::ComponentType;
473 
474  static VISKORES_CONT void save(BinaryBuffer& bb, const BaseType& obj_)
475  {
476  viskores::cont::ArrayHandleSOAStride<ValueType> obj = obj_;
477  for (viskores::IdComponent componentIndex = 0; componentIndex < NUM_COMPONENTS;
478  ++componentIndex)
479  {
480  viskores::cont::ArrayHandleStride<ComponentType> componentArray =
481  obj.GetArray(componentIndex);
482  viskoresdiy::save(bb, componentArray.GetNumberOfValues());
483  viskoresdiy::save(bb, componentArray.GetStride());
484  viskoresdiy::save(bb, componentArray.GetOffset());
485  viskoresdiy::save(bb, componentArray.GetModulo());
486  viskoresdiy::save(bb, componentArray.GetDivisor());
487 
488  viskores::cont::internal::Buffer componentBuffer = componentArray.GetBuffers()[1];
489  bool alreadyWritten = false;
490  for (viskores::IdComponent checkIndex = 0; checkIndex < componentIndex; ++checkIndex)
491  {
492  viskores::cont::internal::Buffer checkBuffer = obj.GetArray(checkIndex).GetBuffers()[1];
493  if (checkBuffer == componentBuffer)
494  {
495  viskoresdiy::save(bb, checkIndex);
496  alreadyWritten = true;
497  break;
498  }
499  }
500  if (!alreadyWritten)
501  {
502  viskoresdiy::save(bb, viskores::IdComponent{ -1 });
503  viskoresdiy::save(bb, componentBuffer);
504  }
505  }
506  }
507 
508  static VISKORES_CONT void load(BinaryBuffer& bb, BaseType& obj_)
509  {
510  viskores::cont::ArrayHandleSOAStride<ValueType> obj;
511  for (std::size_t componentIndex = 0; componentIndex < NUM_COMPONENTS; ++componentIndex)
512  {
513  viskores::Id numValues;
514  viskores::Id stride;
515  viskores::Id offset;
516  viskores::Id modulo;
517  viskores::Id divisor;
518  viskores::cont::internal::Buffer componentBuffer;
519 
520  viskoresdiy::load(bb, numValues);
521  viskoresdiy::load(bb, stride);
522  viskoresdiy::load(bb, offset);
523  viskoresdiy::load(bb, modulo);
524  viskoresdiy::load(bb, divisor);
525 
526  viskores::IdComponent bufferIndex;
527  viskoresdiy::load(bb, bufferIndex);
528  if (bufferIndex < 0)
529  {
530  viskoresdiy::load(bb, componentBuffer);
531  }
532  else
533  {
534  componentBuffer = obj.GetArray(bufferIndex).GetBuffers()[1];
535  }
536 
537  viskores::cont::ArrayHandleStride<ComponentType> componentArray(
538  componentBuffer, numValues, stride, offset, modulo, divisor);
539  obj.SetArray(componentIndex, componentArray);
540  }
541  obj_ = obj;
542  }
543 };
