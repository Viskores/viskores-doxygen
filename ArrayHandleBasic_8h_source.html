284 template <typename T>
285 VISKORES_CONT viskores::cont::ArrayHandleBasic<T> make_ArrayHandle(const T* array,
286  viskores::Id numberOfValues,
287  viskores::CopyFlag copy)
288 {
289  if (copy == viskores::CopyFlag::On)
290  {
291  viskores::cont::ArrayHandleBasic<T> handle;
292  handle.Allocate(numberOfValues);
293  std::copy(array,
294  array + numberOfValues,
295  viskores::cont::ArrayPortalToIteratorBegin(handle.WritePortal()));
296  return handle;
297  }
298  else
299  {
300  return viskores::cont::ArrayHandleBasic<T>(const_cast<T*>(array), numberOfValues, [](void*) {});
301  }
302 }
