56  VISKORES_EXEC viskores::ErrorCode Evaluate(const viskores::Vec3f& point,
57  viskores::FloatDefault& value) const
58  {
59  if (!this->Bounds.Contains(point))
60  return viskores::ErrorCode::CellNotFound;
61 
62  //map world to index space.
63  viskores::Vec3f pointIndex;
64  pointIndex[0] = (point[0] - this->Origin[0]) / this->Spacing[0];
65  pointIndex[1] = (point[1] - this->Origin[1]) / this->Spacing[1];
66  pointIndex[2] = (point[2] - this->Origin[2]) / this->Spacing[2];
67 
68  return this->TriCubicEvaluate(pointIndex, value);
69  }

71 private:
72  VISKORES_EXEC viskores::ErrorCode TriCubicEvaluate(const viskores::Vec3f& pointIndex,
73  viskores::FloatDefault& value) const
74  {
75  viskores::Id nx = this->Dimensions[0], ny = this->Dimensions[1], nz = this->Dimensions[2];
76  viskores::FloatDefault x = pointIndex[0], y = pointIndex[1], z = pointIndex[2];
77  // base integer coords
78  viskores::Id ix = static_cast<viskores::Id>(viskores::Floor(x));
79  viskores::Id iy = static_cast<viskores::Id>(viskores::Floor(y));
80  viskores::Id iz = static_cast<viskores::Id>(viskores::Floor(z));
81 
82  // fractional offsets
83  viskores::FloatDefault tx = x - ix;
84  viskores::FloatDefault ty = y - iy;
85  viskores::FloatDefault tz = z - iz;
86 
87  // Coefficients for tricubic interpolation
88  viskores::FloatDefault P[4 * 4 * 4];
89  viskores::FloatDefault C[4 * 4];
90  viskores::FloatDefault D[4];
91 
92  // 1) Gather 4×4×4 neighborhood into P
93  // P[kk][jj][ii] -> P[(kk*4 + jj)*4 + ii]
94  // data[z0][y0][x0] -> data[(z0*ny + y0)*nx + x0]
95  for (viskores::Id kk = 0; kk < 4; ++kk)
96  {
97  viskores::Id z0 = viskores::Clamp(iz - 1 + kk, 0, nz);
98  for (viskores::Id jj = 0; jj < 4; ++jj)
99  {
100  viskores::Id y0 = viskores::Clamp(iy - 1 + jj, 0, ny);
101  for (viskores::Id ii = 0; ii < 4; ++ii)
102  {
103  viskores::Id x0 = viskores::Clamp(ix - 1 + ii, 0, nx);
104  // flatten 3D (kk,jj,ii) to 1D:
105  viskores::Id pIndex = (kk * 4 + jj) * 4 + ii;
106  // flatten volume coords: (x0,y0,z0) -> 1D index
107  viskores::Id dIndex = (z0 * ny + y0) * nx + x0;
108  P[pIndex] = this->Field.Get(dIndex);
109  }
110  }
111  }
112 
113  // 2) Interpolate in X for each (kk, jj) → C[kk][jj]
114  // C[kk][jj] -> C[kk*4 + jj]
115  for (viskores::Id kk = 0; kk < 4; ++kk)
116  {
117  for (viskores::Id jj = 0; jj < 4; ++jj)
118  {
119  // base offset for P row
120  viskores::Id baseP = (kk * 4 + jj) * 4;
121  viskores::Id cIndex = kk * 4 + jj;
122  C[cIndex] =
123  this->CubicInterpolate(P[baseP + 0], P[baseP + 1], P[baseP + 2], P[baseP + 3], tx);
124  }
125  }
126 
127  // 3) Interpolate in Y for each kk → D[kk]
128  // C[kk][0..3] -> C[kk*4 + 0..3]
129  for (viskores::Id kk = 0; kk < 4; ++kk)
130  {
131  D[kk] =
132  this->CubicInterpolate(C[kk * 4 + 0], C[kk * 4 + 1], C[kk * 4 + 2], C[kk * 4 + 3], ty);
133  }
134 
135  // 4) Interpolate in Z across D[0..3]
136  value = this->CubicInterpolate(D[0], D[1], D[2], D[3], tz);
137  return viskores::ErrorCode::Success;
138  }
139 
140  // 1D cubic‐convolution (Catmull–Rom) kernel
141  // given four samples p0,p1,p2,p3 and relative t in [0,1]
142  VISKORES_EXEC viskores::FloatDefault CubicInterpolate(viskores::FloatDefault p0,
143  viskores::FloatDefault p1,
144  viskores::FloatDefault p2,
145  viskores::FloatDefault p3,
146  viskores::FloatDefault t) const
147  {
148  // Catmull–Rom basis with no tension.
149  viskores::FloatDefault t2 = t * t;
150  viskores::FloatDefault t3 = t2 * t;
151 
152  viskores::FloatDefault m0 = (p2 - p0) * 0.5f;
153  viskores::FloatDefault m1 = (p3 - p1) * 0.5f;
154  viskores::FloatDefault d0 = p1;
155  viskores::FloatDefault d1 = p2;
156 
157  // Hermite form: h00, h10, h01, h11
158  viskores::FloatDefault h00 = 2 * t3 - 3 * t2 + 1;
159  viskores::FloatDefault h10 = t3 - 2 * t2 + t;
160  viskores::FloatDefault h01 = -2 * t3 + 3 * t2;
161  viskores::FloatDefault h11 = t3 - t2;
162 
163  return h00 * d0 + h10 * m0 + h01 * d1 + h11 * m1;
164  }
