61  VISKORES_EXEC viskores::ErrorCode Evaluate(const viskores::Vec3f& point,
62  viskores::FloatDefault& value) const
63  {
64  if (!this->Bounds.Contains(point))
65  return viskores::ErrorCode::CellNotFound;
66 
67  auto x = point[0];
68  auto y = point[1];
69  auto z = point[2];
70 
71  viskores::Id iu = this->FindIndex(this->AxisPortals[0], this->NumX, point[0]);
72  viskores::Id iv = this->FindIndex(this->AxisPortals[1], this->NumY, point[1]);
73  viskores::Id iw = this->FindIndex(this->AxisPortals[2], this->NumZ, point[2]);
74 
75  VISKORES_ASSERT(this->NumX >= 4 && this->NumY >= 4);
76 
77  if (this->NumZ < 4)
78  {
79  // --- bicubic: gather a 4×4 patch in X–Y at single k = clamp(iw,0,ny−1) ---
80  viskores::FloatDefault P2d[4 * 4];
81  for (viskores::Id jj = 0; jj < 4; ++jj)
82  {
83  viskores::Id j = viskores::Clamp(iv - 1 + jj, 0, this->NumY - 1);
84  for (viskores::Id ii = 0; ii < 4; ++ii)
85  {
86  viskores::Id i = viskores::Clamp(iu - 1 + ii, 0, this->NumX - 1);
87  // flatten (i,j, 0) → data index
88  P2d[jj * 4 + ii] = this->Field.Get((j * this->NumX - 1) + i);
89  }
90  }
91  // 3) bicubic along X → C2[4]
92  viskores::FloatDefault C2[4];
93  for (int jj = 0; jj < 4; ++jj)
94  {
95  auto x0 = this->AxisPortals[0].Get(iu - 1);
96  auto x1 = this->AxisPortals[0].Get(iu);
97  auto x2 = this->AxisPortals[0].Get(iu + 1);
98  auto x3 = this->AxisPortals[0].Get(iu + 2);
99  C2[jj] = this->CubicInterpolateNonUniform(
100  x0, x1, x2, x3, P2d[jj * 4 + 0], P2d[jj * 4 + 1], P2d[jj * 4 + 2], P2d[jj * 4 + 3], x);
101  }
102  // 4) bicubic along Y on C2 → final
103  auto y0 = this->AxisPortals[1].Get(iv - 1);
104  auto y1 = this->AxisPortals[1].Get(iv);
105  auto y2 = this->AxisPortals[1].Get(iv + 1);
106  auto y3 = this->AxisPortals[1].Get(iv + 2);
107  value = this->CubicInterpolateNonUniform(y0, y1, y2, y3, C2[0], C2[1], C2[2], C2[3], y);
108  return viskores::ErrorCode::Success;
109  }
110 
111  viskores::FloatDefault P[4 * 4 * 4];
112  for (viskores::Id kk = 0; kk < 4; ++kk)
113  {
114  viskores::Id k = viskores::Clamp(iw - 1 + kk, 0, this->NumZ - 1);
115  for (viskores::Id jj = 0; jj < 4; ++jj)
116  {
117  viskores::Id j = viskores::Clamp(iv - 1 + jj, 0, this->NumY - 1);
118  for (viskores::Id ii = 0; ii < 4; ++ii)
119  {
120  viskores::Id i = viskores::Clamp(iu - 1 + ii, 0, this->NumX - 1);
121  auto pIndex = (kk * 4 + jj) * 4 + ii;
122  auto dIndex = (k * this->NumY + j) * this->NumX + i;
123  P[pIndex] = this->Field.Get(dIndex);
124  }
125  }
126  }
127 
128  // interpolate in X for each (kk,jj) → Cbuf[16]
129  viskores::FloatDefault Cbuf[4 * 4];
130  for (int kk = 0; kk < 4; ++kk)
131  for (int jj = 0; jj < 4; ++jj)
132  {
133  auto x0 = this->AxisPortals[0].Get(iu - 1);
134  auto x1 = this->AxisPortals[0].Get(iu);
135  auto x2 = this->AxisPortals[0].Get(iu + 1);
136  auto x3 = this->AxisPortals[0].Get(iu + 2);
137  auto p0 = P[(kk * 4 + jj) * 4 + 0];
138  auto p1 = P[(kk * 4 + jj) * 4 + 1];
139  auto p2 = P[(kk * 4 + jj) * 4 + 2];
140  auto p3 = P[(kk * 4 + jj) * 4 + 3];
141  Cbuf[kk * 4 + jj] = this->CubicInterpolateNonUniform(x0, x1, x2, x3, p0, p1, p2, p3, x);
142  }
143 
144  // interpolate in Y → D[4]
145  viskores::FloatDefault D[4];
146  for (int kk = 0; kk < 4; ++kk)
147  {
148  auto y0 = this->AxisPortals[1].Get(iv - 1);
149  auto y1 = this->AxisPortals[1].Get(iv);
150  auto y2 = this->AxisPortals[1].Get(iv + 1);
151  auto y3 = this->AxisPortals[1].Get(iv + 2);
152  D[kk] = this->CubicInterpolateNonUniform(
153  y0, y1, y2, y3, Cbuf[kk * 4 + 0], Cbuf[kk * 4 + 1], Cbuf[kk * 4 + 2], Cbuf[kk * 4 + 3], y);
154  }
155 
156  // interpolate in Z
157  auto z0 = this->AxisPortals[2].Get(iw - 1);
158  auto z1 = this->AxisPortals[2].Get(iw);
159  auto z2 = this->AxisPortals[2].Get(iw + 1);
160  auto z3 = this->AxisPortals[2].Get(iw + 2);
161  value = this->CubicInterpolateNonUniform(z0, z1, z2, z3, D[0], D[1], D[2], D[3], z);
162 
163  return viskores::ErrorCode::Success;
164  }

167  VISKORES_EXEC viskores::Id FindIndex(const AxisPortalType& axis,
168  const viskores::Id& N,
169  viskores::FloatDefault val) const
170  {
171  // 1) Binary search for the largest index i with coords[i] <= val
172  viskores::Id left = 0;
173  viskores::Id right = N - 1;
174  while (left <= right)
175  {
176  viskores::Id mid = left + (right - left) / 2;
177  if (axis.Get(mid) <= val)
178  {
179  // mid is still ≤ val, so it might be our i
180  left = mid + 1;
181  }
182  else
183  {
184  // coords[mid] > val, so the index we want is below mid
185  right = mid - 1;
186  }
187  }
188  // when loop ends, `right` is the last index where coords[right] <= val
189  viskores::Id i = right;
190 
191  // 2) Clamp i into [1, N-3]
192  return viskores::Max(viskores::Id(1), viskores::Min(i, N - 3));
193 
194  //return viskores::Clamp(viskores::Id(i), viskores::Id(1), viskores::Id(N - 3));
195  }

197  VISKORES_EXEC
198  viskores::FloatDefault CubicInterpolateNonUniform(viskores::FloatDefault x0,
199  viskores::FloatDefault x1,
200  viskores::FloatDefault x2,
201  viskores::FloatDefault x3,
202  viskores::FloatDefault p0,
203  viskores::FloatDefault p1,
204  viskores::FloatDefault p2,
205  viskores::FloatDefault p3,
206  viskores::FloatDefault x) const
207  {
208  // 1) Compute interval lengths
209  viskores::FloatDefault h0 = x1 - x0;
210  viskores::FloatDefault h1 = x2 - x1;
211  viskores::FloatDefault h2 = x3 - x2;
212  VISKORES_ASSERT(h0 > 0.0f && h1 > 0.0f && h2 > 0.0f);
213 
214 
215  // 2) Compute right‐hand sides for second‐derivative system
216  viskores::FloatDefault rhs1 = 6.0f * ((p2 - p1) / h1 - (p1 - p0) / h0);
217  viskores::FloatDefault rhs2 = 6.0f * ((p3 - p2) / h2 - (p2 - p1) / h1);
218 
219  // 3) Build and solve the 2×2 system:
220  // [2(h0+h1) h1 ][d2_1] = [rhs1]
221  // [ h1 2(h1+h2) ][d2_2] [rhs2]
222  viskores::FloatDefault a11 = 2.0f * (h0 + h1);
223  viskores::FloatDefault a12 = h1;
224  viskores::FloatDefault a21 = h1;
225  viskores::FloatDefault a22 = 2.0f * (h1 + h2);
226  viskores::FloatDefault det = a11 * a22 - a12 * a21;
227  VISKORES_ASSERT(det != 0.0f);
228 
229  viskores::FloatDefault d2_1 = (rhs1 * a22 - a12 * rhs2) / det;
230  viskores::FloatDefault d2_2 = (a11 * rhs2 - rhs1 * a21) / det;
231 
232  // 4) Map x into local parameter t ∈ [0,1] on [x1,x2]
233  viskores::FloatDefault t = (x - x1) / h1;
234 
235  // 5) Hermite form of the natural cubic on [x1, x2]
236  viskores::FloatDefault A = 1.0f - t;
237  viskores::FloatDefault B = t;
238  viskores::FloatDefault h1_sq_6 = h1 * h1 / 6.f;
239  viskores::FloatDefault term1 = (A * A * A - A) * h1_sq_6 * d2_1;
240  viskores::FloatDefault term2 = (B * B * B - B) * h1_sq_6 * d2_2;
241 
242  // 6) Combine the linear and curvature parts
243  return A * p1 + B * p2 + term1 + term2;
244  }
